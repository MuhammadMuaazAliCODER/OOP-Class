Basics of Class and Object

A class is a blueprint/template. An object is an instance of a class created from that blueprint.

Key Terms

  • Class — User-defined data type
  • Object — Instance of a class
  • Data members — Variables inside class
  • Member functions — Functions inside class
  • Access specifiers — public, private, protected
  • this pointer — Points to current object

Access Specifiers

SpecifierAccessible From
publicAnywhere
privateSame class only
protectedClass + subclasses

Basic Class Syntax

class Student {
private:
    string name;
    int age;

public:
    void setName(string n) { name = n; }
    string getName()        { return name; }
    void display() {
        cout << "Name: " << name << ", Age: " << age;
    }
};

int main() {
    Student s1;              // stack object
    s1.setName("Ali");
    s1.display();

    Student* ptr = new Student(); // heap object
    ptr->setName("Sara");
    delete ptr;
}
⚠️ Default access in class is private. In struct, it's public.

};

Shallow vs Deep Copy

⚠️ Shallow copy copies the pointer address — both objects share the same memory → double-delete CRASH!
// ❌ SHALLOW — dangerous with dynamic memory
class Shallow {
public:
    int* data;
    Shallow(int v) { data = new int(v); }
    // compiler-generated copy: copies the pointer address only
};

// ✅ DEEP — correct solution
class Deep {
public:
    int* data;
    Deep(int v) { data = new int(v); }
    Deep(const Deep& o) {
        data = new int(*o.data); // NEW allocation, SAME value
    }
    ~Deep() { delete data; }
};
Shallow CopyDeep Copy
PointerAddress copiedNew memory allocated
Independence❌ Shared memory✅ Independent
RiskDouble-delete crashSafe

Destructor & Dynamic Memory (new/delete)

Destructor = ~ClassName() | No return type | No parameters | Called automatically on destruction
class DynArray {
    int* arr;
    int  size;
public:
    DynArray(int n) : size(n) {
        arr = new int[n];
        cout << "Allocated " << n << " ints\n";
    }
    ~DynArray() {
        delete[] arr;   // free array memory
        cout << "Memory freed\n";
    }
};
OperationSyntaxFree With
Single objectnew Type(val)delete ptr
Arraynew Type[n]delete[] ptr
⚠️ Never use delete for arrays — always delete[]. Set pointer to nullptr after delete.

Rule of Three

If you define any one of these, define all three:

class MyStr {
    char* str;
public:
    MyStr(const char* s) { str = new char[strlen(s)+1]; strcpy(str,s); }
    MyStr(const MyStr& o) { str = new char[strlen(o.str)+1]; strcpy(str,o.str); }
    MyStr& operator=(const MyStr& o) {
        if(this != &o) { delete[] str; str=new char[strlen(o.str)+1]; strcpy(str,o.str); }
        return *this;
    }
    ~MyStr() { delete[] str; }
};

Encapsulation

Bundling data + functions together and restricting direct access. Data is private, accessed only through public getters/setters.

✅ Benefits: Data protection | Input validation | Controlled access | Easier maintenance
class BankAccount {
private:
    double balance;
    string owner;
    bool isValid(double amt) { return amt > 0; }

public:
    BankAccount(string name, double initial) : owner(name), balance(initial) {}
    double getBalance() const { return balance; }

    bool deposit(double amt) {
        if(!isValid(amt)) { cout << "Invalid!\n"; return false; }
        balance += amt; return true;
    }
    bool withdraw(double amt) {
        if(!isValid(amt) || amt > balance) { cout << "Failed!\n"; return false; }
        balance -= amt; return true;
    }
};

int main() {
    BankAccount acc("Ali", 1000);
    // acc.balance = 9999; ← ERROR: private
    acc.deposit(500);
    cout << acc.getBalance(); // 1500
}

Member & Friend Functions

class Box {
    double vol;
public:
    Box(double v) : vol(v) {}
    double getVol();                        // member — declared
    friend void compare(Box&, Box&);      // friend — declared
};
double Box::getVol() { return vol; }       // member — defined outside
void compare(Box& a, Box& b) {           // friend — no Box:: prefix
    cout << (a.vol > b.vol ? "A bigger" : "B bigger");
}

Pass by Value vs Reference

void byVal(int x)        { x *= 2; }   // copy — original unchanged
void byRef(int& x)       { x *= 2; }   // reference — original changed
void byConstRef(const int& x) { /* read only, no copy overhead */ }

int main() {
    int n = 10;
    byVal(n); cout << n; // 10 — unchanged
    byRef(n); cout << n; // 20 — modified!
}

+ and − Operators ▼
class Vec {
public:
    int x, y;
    Vec(int x=0, int y=0) : x(x), y(y) {}
    Vec operator+(const Vec& o) const { return Vec(x+o.x, y+o.y); }
    Vec operator-(const Vec& o) const { return Vec(x-o.x, y-o.y); }
};
<< Stream Operator (Friend) ▼
⚠️ Left operand is ostream — must be friend. Return ostream& to allow chaining.
class Point {
public:
    int x, y;
    Point(int x=0, int y=0) : x(x), y(y) {}
    friend ostream& operator<<(ostream& out, const Point& p) {
        out << "(" << p.x << "," << p.y << ")";
        return out; // return for chaining: cout << p1 << p2
    }
};
== and != Operators ▼
class Student {
public:
    int id; string name;
    Student(int id, string n) : id(id), name(n) {}
    bool operator==(const Student& o) const { return id == o.id; }
    bool operator!=(const Student& o) const { return !(*this == o); }
};
++ Prefix vs Postfix ▼
Trick: Postfix takes a dummy int to distinguish from prefix!
class Counter {
    int n;
public:
    Counter(int n=0) : n(n) {}
    Counter& operator++()      { ++n; return *this; }       // PREFIX
    Counter  operator++(int)  { Counter t=*this; ++n; return t; } // POSTFIX
    int get() const { return n; }
};
= Assignment Operator ▼
⚠️ Always check for self-assignment: if (this != &other)
MyArr& operator=(const MyArr& o) {
    if(this != &o) {       // self-assign guard
        delete[] data;     // free existing
        size = o.size;
        data = new int[size];
        for(int i=0; i<size; i++) data[i] = o.data[i];
    }
    return *this;          // allows a = b = c
}

Static Variables and Functions

Static members belong to the class, not to any single object. One copy shared by all instances.
class Employee {
    string name;
    static int count; // shared by ALL objects
public:
    Employee(string n) : name(n) { count++; }
    ~Employee()                   { count--; }
    static int getCount() { return count; } // static fn
    // static fn cannot access 'name' — no 'this' pointer
};
int Employee::count = 0;  // REQUIRED — define outside class

int main() {
    cout << Employee::getCount(); // 0
    { Employee e1("Ali"), e2("Sara"); cout << Employee::getCount(); } // 2
    cout << Employee::getCount(); // 0 (both destroyed)
}

⚠️ Static functions can ONLY access static members — no this pointer available!

Tracing Output — Constructor/Destructor Order

Constructors: called in order of creation. Destructors: called in reverse (LIFO) order.
class Demo {
    string name;
public:
    Demo(string n) : name(n) { cout << "Ctor: " << name << "\n"; }
    ~Demo()                  { cout << "Dtor: " << name << "\n"; }
};
int main() {
    Demo a("A");
    Demo b("B");
    { Demo c("C"); }  // c destroyed here (scope ends)
    Demo d("D");
}  // d, b, a destroyed (reverse order)
// Expected Output:
Ctor: A
Ctor: B
Ctor: C
Dtor: C   ← scope ended
Ctor: D
Dtor: D
Dtor: B
Dtor: A   ← last created, last destroyed
